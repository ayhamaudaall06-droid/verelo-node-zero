        function renderCards() {

        // Drag and Drop

        let draggedCard = null;


        function drag(e) {

            draggedCard = cards.find(c => c.id === e.target.dataset.cardId);

            e.target.classList.add('dragging');

            e.dataTransfer.effectAllowed = 'move';

        }


        function dragEnd(e) {

            e.target.classList.remove('dragging');

        }


        function allowDrop(e) {

            e.preventDefault();

            e.currentTarget.classList.add('drag-over');

        }


        function dragLeave(e) {

            e.currentTarget.classList.remove('drag-over');

        }


        function drop(e) {

            e.preventDefault();

            e.currentTarget.classList.remove('drag-over');

            

            const newColumn = e.currentTarget.id.replace('col-', '');

            const columnEl = document.querySelector(`[data-column="${newColumn}"]`);

            const wipLimit = parseInt(columnEl.dataset.wip);

            const currentCount = parseInt(document.getElementById(`count-${newColumn}`).textContent);

            

            // Check WIP limit

            if (currentCount >= wipLimit && draggedCard.column !== newColumn) {

                alert(`⚠️ WIP Limit reached for ${newColumn} (${wipLimit}). Complete existing work first.`);

                return;

            }

            

            // Move logic based on your rules

            const allowedMoves = {

                'capture': ['intent'],

                'intent': ['capture', 'product'],

                'product': ['intent', 'config'],

                'config': ['product', 'secure'],

                'secure': ['config', 'fulfill'],

                'fulfill': ['secure']

            };

            

            if (!allowedMoves[draggedCard.column].includes(newColumn)) {

                // For demo, we'll allow it with warning. In production, enforce strictly.

                console.warn(`Non-standard move: ${draggedCard.column} → ${newColumn}`);

            }

            

            // Update card

            draggedCard.column = newColumn;

            

            // Update state based on column

            const stateMap = {

                'capture': 'ACQUIRED',

                'intent': 'QUALIFIED',

            console.log(`[KANBAN] Moved ${draggedCard.id} to ${newColumn} → State: ${draggedCard.state}`);

        }
